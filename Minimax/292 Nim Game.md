2. `Recursion`+`Memory`

   `1348820612`: `overflow` **50 / 60** 个通过测试用例

   ```c++
   class Solution {
   public:
       int dp[1000]={0};
       bool canWinNim(int n) {
           if(dp[n]!=0)return dp[n]==1;
           if(n==0)return false;
           if(n<=3)return true;
           for(int i=1;i<=3;i++){
               if(!canWinNim(n-i)){dp[n]=1;return true;}
           }
           dp[n]=-1;
           return false;
       }
   };
   ```
