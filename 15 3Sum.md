# 15.3Sum

执行用时 : 396 ms, 在3Sum的C++提交中击败了14.88% 的用户
内存消耗 : 18.5 MB, 在3Sum的C++提交中击败了24.54% 的用户

```c++
class Solution {
public:
    vector<vector<int>> threeSum(vector<int>& nums) {
        sort(nums.begin(),nums.end());
        vector<vector<int>> ans;
        if(nums.size()<3)return ans;//测试用例[]
        int i=0;
        while(i<nums.size()-1){
          while(i<nums.size()&&i!=0&&nums[i]==nums[i-1])i++;//测试用例[0*n]
          int j=i+1,k=nums.size()-1;
          cout<<i<<j<<k<<endl;
          while(j<k)
          {
            int s=nums[i]+nums[j]+nums[k];
            if(s>0)
              k--;
            else if(s==0){
              vector<int> cc;
              cout<<i<<j<<k<<endl;
              cc.push_back(nums[i]);
              cc.push_back(nums[j]);
              cc.push_back(nums[k]);
              ans.push_back(cc);
              k--;j++;
              while(j<k&&nums[k]==nums[k+1])k--;//测试用例[0*n]
              while(k>j&&nums[j-1]==nums[j])j++;//测试用例[0*n]
            }else
              j++;
          }
          i++;
        }
      //vector排序
      sort(ans.begin(),ans.end());
      ans.erase(unique(ans.begin(), ans.end()), ans.end());
      return ans;
    }
};
```
