#### 方法一：正则表达式

```python
class Solution:
    def isInteger(self, s:str) -> bool:
        p = re.compile(r'[+-]?\d+')
        m = re.match(p,s)
        if m:
            return re.match(p,s).span() == (0, len(s))
        else:
            return False

    def isFloat(self, s:str) -> bool:
        p = re.compile(r'[+-]?(\d+[.]\d+|\d+[.]|[.]\d+)') 
        m = re.match(p,s)
        # print(m)
        if m:
            return re.match(p,s).span() == (0, len(s))
        else:
            return False

    def isNumber(self, s: str) -> bool:
        '''
        i = 0
        j = len(s) - 1
        while i<=j:
            if s[i] == ' ':
                i += 1 
            elif s[j] == ' ':
                j -= 1
            else:
                break
        if i>j:
            return False
        s = s[i:j+1]
        '''
        arr = s.split()
        # print(arr)
        if len(arr)!=1:
            return False
        s = arr[0]
        
        arr = s.split('e')
        if len(arr) == 1:
            arr = s.split('E')
        if len(arr) == 1:
            return self.isInteger(s) or self.isFloat(s)
        elif len(arr) == 2:
            return (self.isInteger(arr[0]) or self.isFloat(arr[0])) and self.isInteger(arr[1])
        else:
            return False
        

        
```
